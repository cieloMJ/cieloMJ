### 문제 풀이
```
for _ in range(10):
    tc = int(input())
    arr = [list(map(int, input().split())) for _ in range(100)]
    # 가장 큰 값
    max_sum = 0
    # 대각선 값 (정방향, 역방향)
    dia_sum = 0
    dia_sum2 = 0
    
    for i in range(100):
        dia_sum += arr[i][i]
        dia_sum2 += arr[i][99-i]
        # row와 col은 추가의 for문을 돌면서 값 더해줌
        row_sum = col_sum = 0
        for j in range(100):
            row_sum += arr[i][j]
            col_sum += arr[j][i]
        # 일단 row, col, 그리고 max_sum 중에서 큰 값으로 계속 갱신
        max_sum = max(max_sum, row_sum, col_sum)
    #마지막엔 max_sum과 대각선 크기를 비교해서 가장 큰 것을 max_sum으로 계산
    max_sum = max(max_sum, dia_sum, dia_sum2)
    print(f'#{tc} {max_sum}')
```

        if k >= max_sum:
            max_sum = k
    diag1 = sum(arr[i][i] for i in range(100))
    diag2 = sum(arr[i][99 - i] for i in range(100))
    max_sum = max(max_sum, diag1, diag2)
    
    print(f'#{tc} {max_sum}')
